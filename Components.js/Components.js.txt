class AnalyticsCard extends HTMLElement {
    connectedCallback() {
        const platform = this.getAttribute('platform') || 'Spotify';
        const value = this.getAttribute('value') || '0';
        const label = this.getAttribute('label') || '';
        this.innerHTML = `
        <div class="bg-slate-800 p-4 rounded-lg text-center">
            <h3 class="text-white font-bold">${platform}</h3>
            <p class="text-primary-400 font-bold text-2xl">${value}</p>
            <p class="text-slate-400 text-sm">${label}</p>
        </div>`;
    }
}
customElements.define('analytics-card', AnalyticsCard);
